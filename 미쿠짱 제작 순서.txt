4 임시 테이블 생성  

	DROP TABLE IF EXISTS TempTranslation;
   	CREATE TABLE TempTranslation (
		id INT,
     		genreId INT,
	    	genreNameKo VARCHAR(255)
	);
  
     이미 생성되어있다면 , truncate TempTranslation 

5 생성한 테이블 마우스 오른쪽 클릭해서 Import 시작

6 테이블 업데이트

-- 원본 테이블 업데이트
